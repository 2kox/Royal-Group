async function loadGallery() { try { const response = await axios.get(`https://res.cloudinary.com/YOUR_CLOUD_NAME/resources/image/list/gallery.json`); const images = response.data.resources; const galleryDiv = document.getElementById('gallery'); images.forEach(image => { const imgElement = document.createElement('img'); imgElement.src = image.secure_url; imgElement.alt = image.context.alt; // عرض التعليق imgElement.style.width = '200px'; imgElement.style.margin = '10px'; const commentElement = document.createElement('p'); commentElement.textContent = image.context.alt || 'No comment provided'; const container = document.createElement('div'); container.appendChild(imgElement); container.appendChild(commentElement); galleryDiv.appendChild(container); }); } catch (error) { console.error('Error fetching images:', error); } }
